def parseIds()

  arr = []

  node_table = @doc.xpath('//table[@class="table table-hover text-left"]')
  node_rows = node_table.xpath('tr[position()>1]')

  node_rows.xpath('td[4]/a[1]/@href').each do |a|
    id = getShipId(a.text)
    arr.push(id) if id != ""
  end

  return arr
end
