def mapDetails()

  values = @details["values"]
  h = {}
  return h if values.nil?

  MAPPING.each do |map|
    if valid(map, values)
      h[map[KEY]] = values[map[SRC]].public_send(map[FORMAT])
    else
      h[map[KEY]] = ""
    end
  end

  return h
end
